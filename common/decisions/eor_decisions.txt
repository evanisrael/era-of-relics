eor_decision_pandora= {
	title = "Ящик Пандоры"
	picture = "gfx/interface/illustrations/decisions/dec_test.dds"

	major = yes
	
	desc = legendary_decision_desc
	confirm_text = legendary_decision_confirm

  is_shown = {
    NOT = {
      is_target_in_global_variable_list = {
        name = unavailable_unique_decisions
        target = flag:eor_decision_pandora
      }
    }
  }

	cost = {
		gold = 30000
	}

  effect = {
    add_character_modifier = {
			modifier = eor_main_modifier_positive
		}

    trigger_event = {
			id = eor_pandora_event.1000
			days = 10
		}

    add_to_global_variable_list = {
      name = unavailable_unique_decisions
      target = flag:eor_decision_pandora
    }
    set_global_variable = {
      name = eor_decision_pandora
      value = scope:pandora_finder
    }
  }

  ai_potential = {

  }

  ai_will_do = {
    base = 10
  }

  cooldown = {
    days = 1
  }
}
