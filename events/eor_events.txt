#pandora's box

eor_event.1001 = {
	type = character_event
  scope = character

  left_portrait = {
      character = ROOT
			animation = admiration
		}

  option {
    name = legendary_decision_desc

    create_eor_pandora_box_effect = {
	  $OWNER$ = { save_scope_as = owner }
	  set_artifact_rarity_illustrious = yes

	  create_artifact = {
      name = pandora_box
      description = pandora_box_desc
      type = sword 
      visuals = pandora_chest
      wealth = scope:wealth
      quality = scope:quality
      history = {
        type = created_before_history
      }
      save_scope_as = eor_pandora_box
      decaying = yes
      modifier = artifact_monthly_piety_4_modifier
      set_variable = { name = historical_unique_artifact value = yes }
      set_variable = { name = pandora_box_artifact }
      }
    }

    give_artifact = { who = root artifact = pandora_box } 
  }
}
